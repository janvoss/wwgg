  Jahr = c(

    "2004", "2009", "2017", "2020", "2023", "2024"
  ),
  Staaten = c(

    "Albanien, Kroatien",
    "Montenegro",

)

# 2. Ausgabe mit knitr::kable

  • Erst ein Anwendungsfall (nach den Anschlägen des 11. September 2001)
